The @toolsync/builtin package provides a set of built-in tools that are available for use in your Toolsync project.

## Tools available

import tools from '@toolsync/builtin/tools.json';
import { joinPathname } from '@/lib/helpers';

<Cards>

{tools.map(tool => (

<Card
  key={tool.name}
  title={tool.name}
  description={tool.description}
  href={joinPathname(tool.path, '/docs/builtin/')}
/>

))}

</Cards>
